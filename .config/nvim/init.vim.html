 48 " Use tab for trigger completion with characters ahead and navigate.
 49 " NOTE: There's always complete item selected by default, you may want to enable
 50 " no select by `"suggest.noselect": true` in your configuration file.
 51 " NOTE: Use command ':verbose imap <tab>' to make sure tab is not mapped by
 52 " other plugin before putting this into your config.
 53 inoremap <silent><expr> <TAB>
 54       \ coc#pum#visible() ? coc#pum#next(1) :
 55       \ CheckBackspace() ? "\<Tab>" :
 56       \ coc#refresh()
 57 inoremap <expr><S-TAB> coc#pum#visible() ? coc#pum#prev(1) : "\<C-h>"

 64 function! CheckBackspace() abort
 65   let col = col('.') - 1
 66   return !col || getline('.')[col - 1]  =~# '\s'
 67 endfunction
